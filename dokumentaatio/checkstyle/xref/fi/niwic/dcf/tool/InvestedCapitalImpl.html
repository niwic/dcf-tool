1   package fi.niwic.dcf.tool;
2   
3   import fi.niwic.dcf.api.BalanceSheet;
4   import fi.niwic.dcf.api.InvestedCapital;
5   
6   public class InvestedCapitalImpl implements InvestedCapital {
7   
8       private BalanceSheet bs;
9       
10      public InvestedCapitalImpl(BalanceSheet balanceSheet) {
11          bs = balanceSheet;
12      }
13      
14      @Override
15      public long getNetOperatingCapital() {
16          long items[] = {
17              bs.getNonInterestBearingFinancialAssets(),
18              bs.getInventory(),
19              Math.negateExact(bs.getShortTermNonInterestBearingLiabilities())
20          };
21          
22          long sum = 0;
23          for (long item : items) {
24              sum = Math.addExact(sum, item);
25          }
26          
27          return sum;
28      }
29  
30      @Override
31      public long getLongTermAssets() {
32          return bs.getFixedAssets();
33      }
34  
35      @Override
36      public long getLiabilities() {
37          return Math.addExact(
38                  bs.getLongTermLiabilities(),
39                  bs.getShortTermInterestBearingLiabilities()
40          );
41      }
42  
43      @Override
44      public long getEquity() {
45          long items[] = {
46              bs.getBoundEquity(),
47              bs.getPastProfitsOrLoss(),
48              bs.getCurrentPeriodProfitOrLoss()
49          };
50          
51          long sum = 0;
52          for (long item : items) {
53              sum = Math.addExact(sum, item);
54          }
55          
56          return sum;
57      }
58      
59  }
