1   package fi.niwic.dcf.ui;
2   
3   import fi.niwic.dcf.api.CostOfCapital;
4   import fi.niwic.dcf.api.DCFCalculation;
5   import fi.niwic.dcf.api.Period;
6   import fi.niwic.dcf.tool.DCFCalculationImpl;
7   import fi.niwic.dcf.tool.FinancialStatementImpl;
8   import fi.niwic.dcf.tool.PerpetualPeriod;
9   import fi.niwic.dcf.tool.WACC;
10  import javafx.stage.Stage;
11  
12  public class UI {
13  
14      private static String title = "DCF Tool";
15      
16      private Stage primaryStage;
17      
18      private MainScene mainScene;
19      private StartScene startScene;
20      
21      private DCFCalculation calculation;
22      
23      /**
24       * Luo uuden käyttöliittymän ja laskelman joka annetaan käyttöliittymälle.
25       * @param primaryStage javafxn pääpiirtoalue
26       * @throws Exception 
27       */
28      public UI(Stage primaryStage) throws Exception {
29          this.primaryStage = primaryStage;
30          primaryStage.setTitle(title);
31          
32          Period pPeriod = new PerpetualPeriod(new FinancialStatementImpl());
33          CostOfCapital coc = new WACC(pPeriod.getCurrentFinancialStatement().getBalanceSheet().getInvestedCapital());
34          
35          calculation = new DCFCalculationImpl(coc);
36          calculation.setPerpetualPeriod(pPeriod);
37          
38          mainScene = new MainScene(primaryStage, calculation);
39          startScene = new StartScene(primaryStage, mainScene, calculation);
40      }
41      
42      /**
43       * Avaa käyttöliittymän ensimmäisen ruudun.
44       */
45      public void render() {
46          primaryStage.setScene(startScene.scene());
47          primaryStage.show();
48      }
49  
50  }
