1   package fi.niwic.dcf.ui;
2   
3   import javafx.event.ActionEvent;
4   import javafx.geometry.Insets;
5   import javafx.geometry.Pos;
6   import javafx.scene.Scene;
7   import javafx.scene.control.Button;
8   import javafx.scene.control.Label;
9   import javafx.scene.control.TextField;
10  import javafx.scene.layout.GridPane;
11  import javafx.stage.Stage;
12  
13  public class StartScene {
14  
15      private static final String COMPANY_NAME = "Company name:";
16      private static final String FIRST_YEAR = "First year:";
17      
18      private Stage stage;
19      private MainScene mainScene;
20      
21      private GridPane grid;
22      private TextField companyName;
23      private TextField startYear;
24      private Label errorLabel;
25      
26      public StartScene(Stage stage, MainScene mainScene) {
27          this.mainScene = mainScene;
28          this.stage = stage;
29          initialize();
30      }
31      
32      private void initialize() {
33          initializeGrid();
34          companyName = createInput(COMPANY_NAME, 0, 0);
35          startYear = createInput(FIRST_YEAR, 0, 1);
36          errorLabel = createLabel("", 0, 2);
37          createStartButton();
38      }
39      
40      private void initializeGrid() {
41          grid = new GridPane();
42          grid.setAlignment(Pos.CENTER);
43          grid.setPadding(new Insets(10, 10, 10, 10));
44          grid.setVgap(5);
45          grid.setHgap(5);
46      }
47      
48      private void createStartButton() {
49          Button start = new Button("Start");
50          GridPane.setConstraints(start, 1, 2);
51          grid.getChildren().add(start);
52          
53          start.setOnAction(this::onStartButton);
54      }
55      
56      private void onStartButton(ActionEvent e) {
57          try {
58              stage.setScene(mainScene.render(companyName.getText(),
59                              Integer.parseInt(startYear.getText())
60                      )
61              );
62          } catch (NumberFormatException ex) {
63              errorLabel.setText("Invalid start year");
64          }
65      }
66      
67      private TextField createInput(String label, int colIndex, int rowIndex) {
68          createLabel(label, colIndex, rowIndex);
69          return createField(colIndex + 1, rowIndex);
70      }
71      
72      private Label createLabel(String text, int colIndex, int rowIndex) {
73          Label label = new Label(text);
74          GridPane.setConstraints(label, colIndex, rowIndex);
75          grid.getChildren().add(label);
76          
77          return label;
78      }
79      
80      private TextField createField(int colIndex, int rowIndex) {
81          TextField input = new TextField();
82          GridPane.setConstraints(input, colIndex, rowIndex);
83          grid.getChildren().add(input);
84          
85          return input;
86      }
87      
88      public Scene scene() {
89          return new Scene(grid, 350, 200);
90      }
91      
92  }
