8   public class FreeCashFlowCalculationViewModel implements PeriodViewModel {

10      private DCFCalculation dcf;
11      
12      /**
13       * Näkymämalli analyysille.
14       * @param dcf laskelma
15       */
16      public FreeCashFlowCalculationViewModel(DCFCalculation dcf) {
17          this.dcf = dcf;
18      }
19      
20      @Override
21      public ArrayList<PeriodView> get() {
22          ArrayList<PeriodView> viewModel = new ArrayList();
23  
24          PeriodView rows[] = {
25              operatingProfit, taxCosts, interestCostTaxEffect, interestIncomeTaxEffect,
26              otherTaxEffect, noplat, depreciation, grossCashFlow, nwcChange,
27              grossInvestments, operatingCashFlow, nonOperCashFlow, freeCashFlow,
28              discountedFreeCashFlow
29          };
30  
31          for (PeriodView row : rows) {
32              viewModel.add(row);
33          }
34  
35          return viewModel;
36      }
37  
38      private FreeCashFlowCalculation fcf(Period period) {
39          return period.getFreeCashFlowCalculation();
40      }
