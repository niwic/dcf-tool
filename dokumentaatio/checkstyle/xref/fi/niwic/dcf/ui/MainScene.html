1   package fi.niwic.dcf.ui;
2   
3   import javafx.event.ActionEvent;
4   import javafx.geometry.Insets;
5   import javafx.scene.Scene;
6   import javafx.scene.control.Button;
7   import javafx.scene.control.Label;
8   import javafx.scene.control.TableColumn;
9   import javafx.scene.control.TableView;
10  import javafx.scene.layout.HBox;
11  import javafx.scene.layout.VBox;
12  import javafx.scene.text.Font;
13  import javafx.scene.text.FontWeight;
14  
15  public class MainScene {
16  
17      private int currentYear;
18      private TableView table;
19      
20      public Scene render(String companyName, int startYear) {
21          currentYear = startYear;
22          
23          VBox vbox = new VBox();
24          vbox.getChildren().addAll(
25              createMenuBar(companyName),
26              createTable()
27          );
28          
29          return new Scene(vbox);
30      }
31      
32      private TableView createTable() {
33          table = new TableView();
34          TableColumn headings = new TableColumn();
35          headings.setSortable(false);
36          table.getColumns().add(headings);
37          
38          return table;
39      }
40      
41      private HBox createMenuBar(String companyName) {
42          HBox hbox = new HBox();
43          hbox.setPadding(new Insets(10));
44          hbox.setSpacing(10);
45          hbox.setStyle("-fx-background-color: #336699;");
46          
47          hbox.getChildren().addAll(
48                  createCompanyNameLabel(companyName),
49                  createAddYearButton()
50          );
51          
52          return hbox;
53      }
54      
55      private Label createCompanyNameLabel(String name) {
56          Label companyName = new Label(name);
57          Font font = Font.font(
58                  Font.getDefault().getFamily(),
59                  FontWeight.BOLD,
60                  Font.getDefault().getSize() + 1
61          );
62          companyName.setFont(font);
63          
64          return companyName;
65      }
66      
67      private Button createAddYearButton() {
68          Button addYear = new Button("Add year");
69          addYear.setOnAction(this::addYear);
70          
71          return addYear;
72      }
73      
74      private void addYear(ActionEvent e) {
75          Integer nextYear = currentYear + 1;
76          TableColumn newYear = new TableColumn(nextYear.toString());
77          currentYear++;
78          table.getColumns().add(newYear);
79      }
80      
81  }
